缓动动画

example
var tween = new Tween(this.icon, [
    { move: [ { x: [10, 100], y: [30, 300] }, 1000] },
    { to: [ { x: 30 , y: 0 } , 500] },
    { wait: 500 },
    { set: { x: 10,  y: 30 } },
    { call: [this.onFinish, this] },
    { target: target }
])
var tween = new egl.Tween(icon)

    .wait(500)
